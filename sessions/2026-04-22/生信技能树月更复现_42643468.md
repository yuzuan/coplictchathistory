score_cols = [c for c in rna_merged.obs.columns if c.startswith('score_')]
scores_df = rna_merged.obs[score_cols]

# 计算置信度（top1 - top2 的分差）
sorted_scores = scores_df.values.copy()
sorted_scores.sort(axis=1)
top1 = sorted_scores[:, -1]      # 最高分
top2 = sorted_scores[:, -2]      # 第二高分
score_diff = top1 - top2

# 预测的细胞类型（每个细胞选得分最高的）
predicted = scores_df.idxmax(axis=1).str.replace('score_', '')

# 低置信度细胞标记为 'Ambiguous'
low_confidence = score_diff < MIN_SCORE_DIFF
predicted[low_confidence] = 'Ambiguous'

# 保存到 obs
rna_merged.obs['predicted_celltype'] = predicted
rna_merged.obs['annotation_confidence'] = score_diff

cluster_anno = {}
for cl in rna_merged.obs['leiden_0.5'].unique():
    mask = rna_merged.obs['leiden_0.5'] == cl
    votes = rna_merged.obs.loc[mask, 'predicted_celltype']
    
    # 排除 'Ambiguous'，只用高置信度投票
    non_ambig = votes[votes != 'Ambiguous']
    
    if len(non_ambig) > 0:
        top = non_ambig.value_counts()
        cluster_anno[cl] = top.index[0]  # 选择最多票的细胞类型

def enhance_umap_minimal(adata, color='celltype', figsize=(8, 8), size=15,

    fig, ax = plt.subplots(figsize=figsize)

    ax.set_title(f'{color}', fontsize=14)

`UMAP` 上每个点的位置还是原来的表达空间邻域关系，所以即使你把很多细分标签收成 `Tumor/Glia` 和 `Myeloid`，那些本来互相接近、处在连续过渡上的细胞还是会挤在一起。Broad map 只能减少“颜色上的碎裂”，不能把原本就连续的生物状态硬拉开。

在你这套数据里，重叠特别多主要有 4 个原因：

1. `Tumor/Glia` 本身就是个超宽的大类  
你把 `AC-like / MES-like / NPC-like / OPC-like / Oligodendrocyte` 全并进来了，它们内部本来就不是一个紧凑球团，而是一整片连续带。合并后颜色统一了，但几何形状不会因此收缩。

2. `Myeloid` 也是连续谱  
`Monocyte / Macrophage / Microglia / CDC / Neutrophil` 在 GBM 里经常互相过渡，尤其肿瘤边界和炎症区域更明显。所以 broad 后依然会看到 Myeloid 朝 Tumor/Glia 方向“贴边”。

3. GBM 生物学上就存在肿瘤-髓系共塑形  
这不是纯技术噪音。肿瘤细胞状态转变、吞噬相关程序、炎症诱导和瘤旁-肿瘤连续性，都会让表达空间里出现真实的“半混合区域”。

这能给出很有用的标签，但对边界细胞不会像 CNV、reference mapping 或多模态证据那样“硬分箱”。所以边界重叠在当前阶段是正常现象。

- 全局层面已经能看出主要 compartments
