```
# 加载必要的包
library(sva)  # 用于批次校正

# 1. 读取phenodata并设置行名
# 假设数据已加载为 GSE242872_pd 和 GSE190138_pd
# 将第一列设置为行名
rownames(GSE242872_pd) <- GSE242872_pd[, 1]
GSE242872_pd <- GSE242872_pd[, -1]

rownames(GSE190138_pd) <- GSE190138_pd[, 1]
GSE190138_pd <- GSE190138_pd[, -1]

# 2. 添加批次信息
GSE242872_pd$batch <- "GSE242872"
GSE190138_pd$batch <- "GSE190138"

# 3. 按group合并phenodata
# 确保两个数据框有相同的列（至少包含group列）
common_cols <- intersect(names(GSE242872_pd), names(GSE190138_pd))
pd <- rbind(GSE242872_pd[, common_cols], GSE190138_pd[, common_cols])

# 4. 合并表达矩阵
# 假设表达数据为 GSE242872 和 GSE190138
# 找到共同基因
common_genes <- intersect(rownames(GSE242872), rownames(GSE190138))
exp_combined <- cbind(GSE242872[common_genes, ], GSE190138[common_genes, ])

# 确保表达矩阵列名与pd行名一致
exp_combined <- exp_combined[, rownames(pd)]

# 5. 使用ComBat进行批次校正
batch <- pd$batch
mod <- model.matrix(~ group, data = pd)  # 保留group效应

exp <- ComBat(dat = exp_combined, 
              batch = batch, 
              mod = mod, 
              par.prior = TRUE)

# 查看结果
dim(pd)
dim(exp)
table(pd$group)
table(pd$batch)
```

2. **添加批次信息**：为每个数据集添加batch列标识来源

3. **合并phenodata**：找到共同的列，使用`rbind`按行合并

4. **合并表达矩阵**：找到共同基因，使用`cbind`按列合并

5. **批次校正**：使用`sva`包的`ComBat`函数进行批次校正，同时保留group的生物学效应

ids_to_symbol <- function(ids, gtf_mapping = NULL, use_rentrez = FALSE, 
                         verbose = TRUE, mart = NULL) {
    if (length(ids) == 0) {
        return(tibble(id = character(0), symbol = character(0), source = character(0)))
    }
    
    orig_ids <- ids
    mapped <- rep(NA_character_, length(ids))
    source <- rep("original", length(ids))
    
    # RefSeq转换
    ref_idx <- grepl('^[NXA][MRK]_', ids)
    if (any(ref_idx)) {
        if (verbose) cat('🔄 RefSeq -> SYMBOL\n')
        mapped[ref_idx] <- convert_refseq(ids[ref_idx])
        source[ref_idx][!is.na(mapped[ref_idx])] <- "RefSeq"
    }
    
    # ENST转换
    enst_idx <- grepl('^ENST', ids)
    if (any(enst_idx)) {
        if (verbose) cat('🔄 ENST -> SYMBOL\n')
        tmp <- convert_enst(ids[enst_idx], mart = mart)
        mapped[enst_idx] <- tmp
        source[enst_idx][!is.na(tmp)] <- "ENST"
    }
    
    # ENSG转换
    ensg_idx <- grepl('^ENSG', ids)
    if (any(ensg_idx)) {
        if (verbose) cat('🔄 ENSG -> SYMBOL\n')
        tmp <- convert_ensg(ids[ensg_idx])
        mapped[ensg_idx] <- tmp
        source[ensg_idx][!is.na(tmp)] <- "ENSG"
    }
    
    # AceView转换
    ace_idx <- grepl('\\.', ids) & !enst_idx & !ref_idx
    if (any(ace_idx)) {
        if (verbose) cat('🔄 AceView -> SYMBOL\n')
        tmp <- convert_aceview(ids[ace_idx])
        mapped[ace_idx] <- tmp
        source[ace_idx][!is.na(tmp)] <- "AceView"
    }
    
    # GenBank转换(可选)
    genbank_idx <- grepl('^BC\\d+', ids)
    if (any(genbank_idx) && use_rentrez) {
        if (verbose) cat('🔄 GenBank -> SYMBOL\n')
        tmp <- convert_genbank(ids[genbank_idx])
        mapped[genbank_idx] <- tmp
        source[genbank_idx][!is.na(tmp)] <- "GenBank"
    }
    
    # Havana转换
    havana_idx <- grepl('^OTTHUMT', ids)
    if (any(havana_idx)) {
        if (verbose) cat('🔄 Havana -> SYMBOL\n')
        tmp <- convert_havana(ids[havana_idx], gtf_mapping = gtf_mapping, mart = mart)
        mapped[havana_idx] <- tmp
        source[havana_idx][!is.na(tmp)] <- "Havana"
    }
    
    # UCSC转换
    ucsc_idx <- grepl('^uc\\d+', ids, ignore.case = TRUE)
    if (any(ucsc_idx)) {
        if (verbose) cat('🔄 UCSC -> SYMBOL\n')
        tmp <- convert_ucsc(ids[ucsc_idx])
        mapped[ucsc_idx] <- tmp
        source[ucsc_idx][!is.na(tmp)] <- "UCSC"
    }
    
    # 保留未转换的原ID
    na_idx <- is.na(mapped)
    mapped[na_idx] <- orig_ids[na_idx]
    
    tibble(
        ID = orig_ids,
        SYMBOL = mapped,
        Source = source
    )
}

比较方向 `experimental vs control` 的含义：

| logFC | 含义 |
|-------|------|
| logFC > 0 | 基因在 **experimental** 中表达 **高于** control → **上调 (Up)** |
| logFC < 0 | 基因在 **experimental** 中表达 **低于** control → **下调 (Down)** |

所以：
- **Up** = experimental 组相对于 control 组上调
- **Down** = experimental 组相对于 control 组下调

这是标准的差异分析比较方式：**实验组 vs 对照组**，对照组作为参考基线。
