n_per_group = 50

# 从差异基因结果中提取top基因
top_genes_unique = []
if key in adata.uns:
    try:
        # 使用 sc.get.rank_genes_groups_df 获取结果
        for g in adata.obs['group'].unique():
            if isinstance(g, str):
                try:
                    dfg = sc.get.rank_genes_groups_df(adata, group=g, key=key)
                    top_genes_unique.extend(dfg['names'].tolist()[:n_per_group])
                except Exception:
                    pass
    except Exception:
        pass

# 如果没有提取到基因，尝试从保存的CSV文件中读取
if len(top_genes_unique) == 0:
    try:
        deg_df = pd.read_csv('write/DEG_migraine_vs_healthy.csv')
        top_genes_unique = deg_df['names'].unique().tolist()[:n_per_group]
    except Exception:
        print("无法读取差异基因结果，请先运行差异分析")
        top_genes_unique = []

# 去重并保留顺序
seen = set()
top_genes_final = [x for x in top_genes_unique if not (x in seen or seen.add(x))]
top_genes_final = top_genes_final[:min(50, len(top_genes_final))]

if len(top_genes_final) == 0:
    print("没有找到有效的差异基因")
else:
    print(f"使用 {len(top_genes_final)} 个差异基因绘制热图")
    
    # 计算每组的平均表达
    use_raw = hasattr(adata, 'raw') and adata.raw is not None
    if use_raw:
        expr_matrix = adata.raw[:, top_genes_final].X
        if hasattr(expr_matrix, 'toarray'):
            expr_matrix = expr_matrix.toarray()
    else:
        expr_matrix = adata[:, top_genes_final].X
        if hasattr(expr_matrix, 'toarray'):
            expr_matrix = expr_matrix.toarray()
    
    # 创建表达数据框
    expr_df = pd.DataFrame(expr_matrix, index=adata.obs_names, columns=top_genes_final)
    expr_df['group'] = adata.obs['group'].values
    
    # 按组计算平均表达
    group_means = expr_df.groupby('group')[top_genes_final].mean()
    
    # 转置为基因x组的矩阵（行=基因，列=组）
    heatmap_data = group_means.T
    
    # 按基因进行z-score标准化
    heatmap_data_z = heatmap_data.sub(heatmap_data.mean(axis=1), axis=0).div(heatmap_data.std(axis=1).replace(0, 1e-9), axis=0)

        heatmap_data_z,

        cbar_kws={'label': 'Z-score', 'shrink': 0.8},
