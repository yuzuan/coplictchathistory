def qc_filter_human(adata, min_counts=500, max_genes=6000, pct_mt=15):
    import numpy as np
    from scipy import sparse
    sc.pp.filter_cells(adata, min_genes=200)

        n_genes_by_counts = np.asarray((x > 0).sum(axis=1)).ravel()

    else:
        x = np.asarray(x, dtype=np.float32)
        total_counts = x.sum(axis=1)
        n_genes_by_counts = (x > 0).sum(axis=1)
        mt_mask = adata.var["mt"].values
        mt_counts = x[:, mt_mask].sum(axis=1)
    adata.obs["total_counts"] = total_counts
    adata.obs["n_genes_by_counts"] = n_genes_by_counts

        & adata.obs["n_genes_by_counts"].between(200, max_genes)

  import numpy as np
  import pandas as pd
  from scipy import sparse

  def adaptive_qc(adata, mt_cap=20, nmad=3, max_genes_cap=12000):
      # 预计算 mt、total_counts、n_genes_by_counts
      adata.var["mt"] = adata.var_names.str.upper().str.startswith("MT-")
      x = adata.X.tocsr() if sparse.issparse(adata.X) else np.asarray(adata.X)
      if sparse.issparse(x):
          total_counts = np.asarray(x.sum(axis=1)).ravel()
          n_genes = np.asarray((x > 0).sum(axis=1)).ravel()
          mt_counts = np.asarray(x[:, adata.var["mt"].values].sum(axis=1)).ravel()
      else:
          total_counts = x.sum(axis=1)
          n_genes = (x > 0).sum(axis=1)
          mt_counts = x[:, adata.var["mt"].values].sum(axis=1)
      adata.obs["total_counts"] = total_counts
      adata.obs["n_genes_by_counts"] = n_genes
      adata.obs["pct_counts_mt"] = (mt_counts / np.maximum(total_counts, 1)) * 100

      keep_mask = np.zeros(adata.n_obs, dtype=bool)
      for sample in adata.obs["sample"].unique():
          sub = adata.obs.loc[adata.obs["sample"] == sample]
          # 基因数上下界：median ± nmad*MAD，再钳制到 [200, max_genes_cap]
          g_med = sub["n_genes_by_counts"].median()
          g_mad = (sub["n_genes_by_counts"] - g_med).abs().median()
          g_low = max(200, g_med - nmad * g_mad)
          g_high = min(max_genes_cap, g_med + nmad * g_mad)

          # UMI 下界：median - nmad*MAD，至少 500
          c_med = sub["total_counts"].median()
          c_mad = (sub["total_counts"] - c_med).abs().median()
          c_low = max(500, c_med - nmad * c_mad)

          # 线粒体上界：min(99分位数, median+ nmad*MAD, mt_cap)
          mt_med = sub["pct_counts_mt"].median()
          mt_mad = (sub["pct_counts_mt"] - mt_med).abs().median()
          mt_high = min(sub["pct_counts_mt"].quantile(0.99), mt_med + nmad * mt_mad, mt_cap)

          keep_mask |= (
              (adata.obs["sample"] == sample)
              & (adata.obs["n_genes_by_counts"].between(g_low, g_high))
              & (adata.obs["total_counts"] >= c_low)
              & (adata.obs["pct_counts_mt"] <= mt_high)
          )
      return adata[keep_mask].copy()
