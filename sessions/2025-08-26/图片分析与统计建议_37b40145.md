# 将数值格式化为 "HR (low-upper)" 与 P 值字符串
fmt_hr <- function(hr, low, up) {
  sprintf("%.2f (%.2f-%.2f)", hr, low, up)
}
fmt_p <- function(p) {
  if (is.na(p)) return(NA_character_)
  if (p < 0.001) return("<0.001")
  sprintf("%.3f", p)
}

# 构建一张单独的 forestplot（含左侧表格）
make_fp <- function(df, title_text = "") {
  # df: data.frame with columns: Variable, HR, Lower, Upper, P
  n <- nrow(df)
  # table text: header + rows
  tabletext <- rbind(
    c("Characteristic", "HR (95% CI)", "P value"),
    cbind(df$Variable,
          sapply(seq_len(n), function(i) fmt_hr(df$HR[i], df$Lower[i], df$Upper[i])),
          sapply(df$P, fmt_p))
  )
  # numeric vectors for plot (include NA for header row)
  mean  <- c(NA, df$HR)
  lower <- c(NA, df$Lower)
  upper <- c(NA, df$Upper)
  # create the forestplot grob
  gp <- fpTxtGp(label = gpar(fontfamily = "", cex = 0.9),
                ticks = gpar(cex=0.8),
                xlab = gpar(cex = 0.9))
  # Colors & box size
  col <- fpColors(box = "darkred", lines = "darkred", summary = "black")
  # Draw plot (we will capture as grob)
  # Use xlog=TRUE to present HR on log scale with center line at 1
  p <- forestplot(labeltext = tabletext,
                  mean = mean,
                  lower = lower,
                  upper = upper,
                  is.summary = c(TRUE, rep(FALSE, n)),
                  zero = 1,
                  xlog = TRUE,
                  col = col,
                  graph.pos = 4,
                  txt_gp = gp,
                  hrzl_lines = list("1" = gpar(lty = 1, col = "#444444")),
                  title = title_text,
                  xticks = c(0.25, 0.5, 1, 2, 4, 8),
                  boxsize = rep(0.12, length(mean))
  )
  # forestplot returns a "gtable" that can be drawn; wrap it into a ggdrawable object for cowplot
  return(p)
}

# Helper to convert forestplot object to grob for cowplot
fp_to_grob <- function(fp_obj) {
  # fp_obj is the result returned by forestplot()
  # draw it to a null device and captureGrob
  grid.newpage()
  gp <- grid.grabExpr(draw(fp_obj))
  return(gp)
}

# 构建一张单独的 forestplot（含左侧表格）
make_fp <- function(df, title_text = "") {
  # df: data.frame with columns: Variable, HR, Lower, Upper, P
  n <- nrow(df)
  # table text: header + rows
  tabletext <- rbind(
    c("Characteristic", "HR (95% CI)", "P value"),
    cbind(df$Variable,
          sapply(seq_len(n), function(i) fmt_hr(df$HR[i], df$Lower[i], df$Upper[i])),
          sapply(df$P, fmt_p))
  )
  # numeric vectors for plot (include NA for header row)
  mean  <- c(NA, df$HR)
  lower <- c(NA, df$Lower)
  upper <- c(NA, df$Upper)
  # create the forestplot grob
  gp <- fpTxtGp(label = gpar(fontfamily = "", cex = 0.9),
                ticks = gpar(cex=0.8),
                xlab = gpar(cex = 0.9))
  # Colors & box size
  col <- fpColors(box = "darkred", lines = "darkred", summary = "black")
  # Draw plot (we will capture as grob)
  # Use xlog=TRUE to present HR on log scale with center line at 1
  p <- forestplot(labeltext = tabletext,
                  mean = mean,
                  lower = lower,
                  upper = upper,
                  is.summary = c(TRUE, rep(FALSE, n)),
                  zero = 1,
                  xlog = TRUE,
                  col = col,
                  graph.pos = 4,
                  txt_gp = gp,
                  hrzl_lines = list("1" = gpar(lty = 1, col = "#444444")),
                  title = title_text,
                  xticks = c(0.25, 0.5, 1, 2, 4, 8),
                  boxsize = rep(0.12, length(mean))
  )
  # forestplot returns a "gtable" that can be drawn; wrap it into a ggdrawable object for cowplot
  return(p)
}
