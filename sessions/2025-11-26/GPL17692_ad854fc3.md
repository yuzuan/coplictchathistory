我看到这个 CSV 文件包含 `ID` 和 `SYMBOL` 列。`SYMBOL` 列看起来包含的是转录本 ID（如 NR_、NM_、ENST00000 等），而不是标准的基因符号。需要将这些转录本 ID 转换为基因符号。

以下是 R 代码，用于将 `SYMBOL` 列中的转录本 ID 转换为基因符号：

```r
# 安装所需包（如果尚未安装）
if (!requireNamespace("BiocManager", quietly = TRUE)) {
  install.packages("BiocManager")
}
if (!requireNamespace("biomaRt", quietly = TRUE)) {
  BiocManager::install("biomaRt")
}
if (!requireNamespace("org.Hs.eg.db", quietly = TRUE)) {
  BiocManager::install("org.Hs.eg.db")
}

# 加载包
library(biomaRt)
library(org.Hs.eg.db)
library(dplyr)

# 设置工作目录
setwd("/Users/yuzuan/Desktop/2025_08_10 fastGEO/test/GPL17692")

# 读取数据
annotation <- read.csv("GPL17692_annotation.csv", stringsAsFactors = FALSE)

# 查看数据结构
head(annotation)

# 连接到 Ensembl 数据库
ensembl <- useMart("ensembl", dataset = "hsapiens_gene_ensembl")

# 提取不同类型的转录本 ID
# 1. ENST 开头的是 Ensembl 转录本 ID
enst_ids <- annotation$SYMBOL[grepl("^ENST", annotation$SYMBOL)]

# 2. NM_ 和 NR_ 开头的是 RefSeq ID
refseq_ids <- annotation$SYMBOL[grepl("^(NM_|NR_|XR_)", annotation$SYMBOL)]

# 转换 Ensembl 转录本 ID 到基因符号
if (length(enst_ids) > 0) {
  enst_to_gene <- getBM(
    attributes = c("ensembl_transcript_id", "hgnc_symbol"),
    filters = "ensembl_transcript_id",
    values = enst_ids,
    mart = ensembl
  )
  colnames(enst_to_gene) <- c("SYMBOL", "gene")
}

# 转换 RefSeq ID 到基因符号
if (length(refseq_ids) > 0) {
  refseq_to_gene <- getBM(
    attributes = c("refseq_mrna", "hgnc_symbol"),
    filters = "refseq_mrna",
    values = refseq_ids,
    mart = ensembl
  )
  colnames(refseq_to_gene) <- c("SYMBOL", "gene")
  
  # 也查询 RefSeq ncRNA
  refseq_ncrna_to_gene <- getBM(
    attributes = c("refseq_ncrna", "hgnc_symbol"),
    filters = "refseq_ncrna",
    values = refseq_ids,
    mart = ensembl
  )
  colnames(refseq_ncrna_to_gene) <- c("SYMBOL", "gene")
  
  # 合并 RefSeq 结果
  refseq_to_gene <- rbind(refseq_to_gene, refseq_ncrna_to_gene)
}

# 合并所有转换结果
gene_mapping <- data.frame(SYMBOL = character(), gene = character())
if (exists("enst_to_gene")) {
  gene_mapping <- rbind(gene_mapping, enst_to_gene)
}
if (exists("refseq_to_gene")) {
  gene_mapping <- rbind(gene_mapping, refseq_to_gene)
}

# 去重（保留第一个匹配）
gene_mapping <- gene_mapping[!duplicated(gene_mapping$SYMBOL), ]

# 将基因符号合并到原始数据
result <- annotation %>%
  left_join(gene_mapping, by = "SYMBOL") %>%
  select(ID, SYMBOL, gene)

# 将空值和 "---" 的 gene 设为 NA
result$gene[result$SYMBOL == "---"] <- NA
result$gene[result$gene == ""] <- NA

# 查看结果
head(result, 20)

# 保存结果
write.csv(result, "GPL17692_annotation_with_gene.csv", row.names = FALSE)

# 统计转换成功率
total <- nrow(result)
converted <- sum(!is.na(result$gene) & result$gene != "")
cat("总探针数:", total, "\n")
cat("成功转换:", converted, "\n")
cat("转换率:", round(converted/total * 100, 2), "%\n")
```

这段代码会：

1. **读取**

文件
2. **识别**不同类型的转录本 ID（ENST、NM_、NR_、XR_ 等）
3. **使用 biomaRt** 连接 Ensembl 数据库，将转录本 ID 转换为 HGNC 基因符号
4. **保留** `ID` 和 `SYMBOL` 列，新增 `gene` 列
5. **保存**结果到 `GPL17692_annotation_with_gene.csv`
