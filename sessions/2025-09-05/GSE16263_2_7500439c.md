解决方法如下：

2. 如果你想复制整个目录（包括目录下所有文件），应该用 shutil.copytree。

针对你的需求（把每个样本文件夹下的文件移动到目标文件夹），推荐如下代码：

import shutil
import os

folder = "GSE162631_RAW"
base_dir = "GSE162631"

# 只获取样本文件夹（目录），不包括文件
samples = [f for f in os.listdir(folder) if os.path.isdir(os.path.join(folder, f))]

for sample in samples:
    src_dir = os.path.join(folder, sample)
    dest_dir = os.path.join(base_dir, sample)
    os.makedirs(dest_dir, exist_ok=True)
    # 遍历每个样本文件夹下的文件
    for file in os.listdir(src_dir):
        src_file = os.path.join(src_dir, file)
        if os.path.isfile(src_file):
            shutil.copy(src_file, dest_dir)
