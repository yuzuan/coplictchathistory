convert_enst <- function(ids) {
  ids_clean <- sub("\\.\\d+$", "", ids) # 去掉版本号
  
  res <- tryCatch({
    suppressMessages({
      library(clusterProfiler)
      bitr(ids_clean, fromType = "ENSEMBLTRANS", 
           toType = "SYMBOL", OrgDb = org.Hs.eg.db)
    })
  }, error = function(e) NULL)
  
  mapped <- rep(NA, length(ids))
  names(mapped) <- ids
  
  if (!is.null(res) && nrow(res) > 0) {
    mapped[ids %in% res$ENSEMBLTRANS] <- res$SYMBOL[match(ids_clean, res$ENSEMBLTRANS)]
  }
  
  # fallback: org.Hs.eg.db
  na_idx <- is.na(mapped)
  if (any(na_idx)) {
    mapped[na_idx] <- mapIds(org.Hs.eg.db,
                             keys = ids_clean[na_idx],
                             keytype = "ENSEMBLTRANS",
                             column = "SYMBOL",
                             multiVals = "first")
  }
  
  mapped
}

```r
ids_to_symbol <- function(ids, gtf_mapping = NULL, use_rentrez = FALSE, verbose = TRUE, mart = NULL) {
  if (length(ids) == 0) return(character(0))
  orig_ids <- ids
  mapped <- setNames(rep(NA_character_, length(ids)), ids)

  # 逐类匹配（每步仅处理仍未映射的项）
  remaining <- which(is.na(mapped))

  # RefSeq
  ref_idx <- grepl('^[NXA][MRK]_', ids)
  if (any(ref_idx)) {
    if (verbose) cat('RefSeq -> SYMBOL\n')
    mapped[ref_idx] <- convert_refseq(ids[ref_idx])
  }

  remaining <- which(is.na(mapped))
  # ENST
  enst_idx <- grepl('^ENST', ids)
  if (any(enst_idx)) {
    if (verbose) cat('ENST -> SYMBOL\n')
    mapped[enst_idx] <- convert_enst(ids[enst_idx], mart = mart)
  }

  remaining <- which(is.na(mapped))
  # ENSG
  ensg_idx <- grepl('^ENSG', ids)
  if (any(ensg_idx)) {
    if (verbose) cat('ENSG -> SYMBOL\n')
    mapped[ensg_idx] <- convert_ensg(ids[ensg_idx])
  }

  remaining <- which(is.na(mapped))
  # AceView
  ace_idx <- grepl('\\.', ids) & !enst_idx & !ref_idx
  if (any(ace_idx)) {
    if (verbose) cat('AceView -> SYMBOL (点前部分)\n')
    mapped[ace_idx] <- convert_aceview(ids[ace_idx])
  }

  remaining <- which(is.na(mapped))
  # GenBank
  genbank_idx <- grepl('^BC\\d+', ids)
  if (any(genbank_idx) && use_rentrez) {
    if (verbose) cat('GenBank (rentrez) -> SYMBOL\n')
    mapped[genbank_idx] <- convert_genbank(ids[genbank_idx])
  }

  remaining <- which(is.na(mapped))
  # Havana
  havana_idx <- grepl('^OTTHUMT', ids)
  if (any(havana_idx)) {
    if (verbose) cat('Havana -> SYMBOL\n')
    mapped[havana_idx] <- convert_havana(ids[havana_idx], gtf_mapping = gtf_mapping, mart = mart)
  }

  remaining <- which(is.na(mapped))
  # UCSC
  ucsc_idx <- grepl('^uc\\d+', ids, ignore.case = TRUE)
  if (any(ucsc_idx)) {
    if (verbose) cat('UCSC TXID -> SYMBOL\n')
    mapped[ucsc_idx] <- convert_ucsc(ids[ucsc_idx])
  }

  # 兜底：保留原 id
  mapped[is.na(mapped)] <- names(mapped)[is.na(mapped)]
  mapped
}
```

简短检查与建议（要点）：
- 有多处直接使用 adata.obs['celltype'].cat 或 .cat.categories，未保证该列为 pandas.Categorical，可能触发 AttributeError。
- 早期创建 palette 的地方用到了 .cat，需改为“安全获取 categories”再生成调色板。
- sc.pl.umap(..., split='group') 在部分 scanpy/matplotlib 版本会报错（PathCollection.set() got unexpected keyword argument 'split'），建议用 try/except 回退到手动分面绘图。
- DotPlot 中以大小写匹配基因时没有把匹配结果映射回真实的 adata.var_names，可能导致传入不存在的名字或顺序问题。
- 若数据矩阵为稀疏矩阵，直接 .X.toarray() 可能占内存，必要时可用 .X.A 或 sc.get.obs_vector。

问题（简短）：sc.pl.umap 报 KeyError 是因为 marker_genes 中的基因名在 adata.var_names（或 adata.raw.var_names）中找不到（大小写/符号不同）。解决：先把候选基因映射到数据里真实的基因名（兼容大小写、优先 adata.var，然后 fallback 到 adata.raw），再传给 sc.pl.umap；同时限制数量并保存图。

请用下面代码替换你 notebook 中绘制 UMAP 那个单元（文件已给出路径）。运行后会打印匹配情况并绘图/保存。

````python

# ...existing code...
# UMAP上绘制这些基因（兼容大小写，优先 adata.var，再 fallback 到 adata.raw）
marker_genes = ['MKI67','TOP2A','EPCAM','KRT19','KRT7','DCN','LUM','GSN','PECAM1','VWF','CD68','CD163','CD14','TPSAB1','TPSB2','C1QA','C1QB','S100A9','S100A8','GNLY','GZMA','NKG7','GZMK','CD3D','CD3E','CD4','CD8A','CD19','CD79A','MS4A1','IGHG1','MZB1','SDC1']

if adata is None:
    print('adata 未加载，跳过绘图')
else:
    # 建立大写->真实名映射（var 优先，raw 为备选）
    var_map = {name.upper(): name for name in adata.var_names}
    raw_map = {}
    if getattr(adata, 'raw', None) is not None:
        raw_map = {name.upper(): name for name in adata.raw.var_names}
    combined_map = {**raw_map, **var_map}  # var_map 覆盖 raw_map

    genes_present = [combined_map[g.upper()] for g in marker_genes if g.upper() in combined_map]
    missing = [g for g in marker_genes if g.upper() not in combined_map]

    print(f'匹配到 {len(genes_present)} 个基因，缺失 {len(missing)} 个（示例缺失: {missing[:10]}）')

    if len(genes_present) == 0:
        print('未找到任何基因，检查 adata.var_names / adata.raw.var_names 的命名方式')
    else:
        # 限制绘图数量以免太拥挤
        genes_to_plot = genes_present[:24]
        use_raw_flag = getattr(adata, 'raw', None) is not None and any(g in adata.raw.var_names for g in genes_to_plot)

        try:
            sc.pl.umap(adata, color=genes_to_plot, ncols=4, cmap='viridis', use_raw=use_raw_flag, show=False)
            plt.savefig('./figures/marker_genes_umap.png', bbox_inches='tight', dpi=150)
            plt.close()
            print('已保存 ./figures/marker_genes_umap.png')
        except Exception as e:
            print('绘图出错:', e)
````
