library(hdf5r)

setwd("/Users/yuzuan/Downloads/生信技能树月更复现/肿瘤及瘤旁 2/write")

# 直接读取 h5ad 文件
h5_file <- H5File$new("adata_for_R.h5ad", mode = "r")

# 读取表达矩阵
X <- h5_file[["X"]]
if (inherits(X, "H5Group")) {
  # 稀疏矩阵格式
  data <- X[["data"]]$read()
  indices <- X[["indices"]]$read()
  indptr <- X[["indptr"]]$read()
  shape <- X$attr_open("shape")$read()
  
  library(Matrix)
  counts <- sparseMatrix(
    i = indices + 1,
    p = indptr,
    x = data,
    dims = shape,
    index1 = TRUE
  )
} else {
  # 密集矩阵
  counts <- t(X$read())
}

# 读取细胞和基因名称
obs_names <- h5_file[["obs"]]$attr_open("_index")$read()
var_names <- h5_file[["var"]]$attr_open("_index")$read()

rownames(counts) <- var_names
colnames(counts) <- obs_names

# 读取metadata
obs_keys <- names(h5_file[["obs"]])
metadata <- data.frame(row.names = obs_names)
for (key in obs_keys) {
  if (key != "_index") {
    metadata[[key]] <- h5_file[["obs"]][[key]]$read()
  }
}

# 创建 Seurat 对象
sce.all.int <- CreateSeuratObject(counts = counts, meta.data = metadata)

# 添加降维结果
if ("X_pca" %in% names(h5_file[["obsm"]])) {
  pca_embeddings <- h5_file[["obsm/X_pca"]]$read()
  rownames(pca_embeddings) <- obs_names
  colnames(pca_embeddings) <- paste0("PC_", 1:ncol(pca_embeddings))
  sce.all.int[["pca"]] <- CreateDimReducObject(
    embeddings = pca_embeddings,
    key = "PC_",
    assay = "RNA"
  )
}

if ("X_umap" %in% names(h5_file[["obsm"]])) {
  umap_embeddings <- h5_file[["obsm/X_umap"]]$read()
  rownames(umap_embeddings) <- obs_names
  colnames(umap_embeddings) <- paste0("UMAP_", 1:ncol(umap_embeddings))
  sce.all.int[["umap"]] <- CreateDimReducObject(
    embeddings = umap_embeddings,
    key = "UMAP_",
    assay = "RNA"
  )
}

h5_file$close_all()

# 查看结果
print(sce.all.int)
DimPlot(sce.all.int, reduction = "umap")
