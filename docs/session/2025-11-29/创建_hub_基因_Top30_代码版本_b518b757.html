# ============================================
# 加载R包

library(ggplot2)    
library(dplyr)      
library(purrr)      
library(readxl)     
library(sf)         
library(ggspatial)  

# ============================================
# 准备数据
# ============================================

# 读取中国省级分区数据
china_province <- st_read("china_province.geojson", quiet = TRUE) %>%  
  st_make_valid() %>%            
  st_transform(4326)            

# 读取中国市级分区数据
china_city <- st_read("china_city.geojson", quiet = TRUE) %>%  
  st_make_valid() %>%              
  st_transform(4326)    

# 读入样本数据
data <- read_excel("data2.xlsx")

# 对数值数据进行分类
data <- data %>%
  mutate(
    Level = case_when(
      Value < 50 ~ "< 50",
      Value >= 50 & Value < 100 ~ "50 - 100",
      Value >= 100 & Value < 300 ~ "100 - 300",
      Value >= 300 & Value < 650 ~ "300 - 650",
      TRUE ~ "other" 
    )
  )

# 将分类变量转换为因子并设置顺序
data$Level <- factor(data$Level, levels = c("< 50", "50 - 100", "100 - 300", "300 - 650"))

# ============================================
# 坐标转换
# ============================================

# 定义等距方位投影（以中国中心为基准点）
aeqd_crs <- "+proj=aeqd +lat_0=35 +lon_0=105 +units=km"

# 城市坐标转换
city_sf <- china_city %>%
  filter(name %in% data$City) %>%  
  st_centroid() %>%                
  st_transform(aeqd_crs)           

# 提取城市坐标并与数据合并
city_data <- st_coordinates(city_sf) %>%  
  as.data.frame() %>%                     
  cbind(City = city_sf$name) %>%          
  left_join(data, by = "City") %>%        
  filter(!is.na(Value) & Value > 0)       

# 省份坐标转换
province_sf <- china_province %>%
  st_centroid() %>%              
  st_transform(aeqd_crs)         

# 提取省份中心点坐标
province_centroids <- st_coordinates(province_sf) %>%
  as.data.frame() %>%                     
  cbind(Province = province_sf$Province)  

# ============================================
# 绑图 - 基础地图
# ============================================
p1 <- ggplot() +
  geom_sf(data = china_province, fill = NA, color = "gray60", linewidth = 0.3) +      
  geom_sf(data = north_south_line, aes(color = "north_south_line"), linewidth = 1) +  
  coord_sf(crs = aeqd_crs,                   
           xlim = c(-2700, 2400),            
           ylim = c(-2000, 2400),             
           expand = FALSE) +                  
  scale_color_manual(name = NULL,             
                     values = c("north_south_line" = "#467f47"),  
                     labels = c("north_south_line" = "North and South China devision line"),
                     guide = guide_legend(override.aes = list(linewidth = 1), order = 2)) + 
  annotation_scale(location = "bl", width_hint = 0.1, style = "ticks") +
  annotation_north_arrow(location = "tr", which_north = TRUE,             
                         height = unit(1, "cm"), width = unit(1, "cm"),
                         style = north_arrow_fancy_orienteering)

# ============================================
# 绑图 - 气泡图
# ============================================
p2 <- p1 +
  geom_point(data = city_data,
             aes(x = X, y = Y, size = Level),              
             color = "#cd302d", fill = "#df979b",          
             shape = 21, stroke = 0.3) +                   
  scale_size_manual(
    name = "City-level groundwater depletion\n(million m³)",  
    values = c("< 50" = 2.5,        
               "50 - 100" = 4,
               "100 - 300" = 5,
               "300 - 650" = 7)) +
  geom_text(data = province_centroids, aes(x = X, y = Y, label = Province),  
            size = 4, color = "grey15")

# ============================================
# 绑图 - 缩略图
# ============================================
p3 <- ggplot() +
  geom_sf(data = china_province, fill = NA, color = "gray60", linewidth = 0.3) +
  coord_sf(crs = aeqd_crs,                   
           xlim = c(0, 1850),               
           ylim = c(-3700, -1000),           
           expand = FALSE) +                 
  theme_bw() +                                
  theme(plot.background = element_blank(),
        panel.grid = element_blank(),
        axis.text = element_blank(),
        axis.ticks = element_blank())

# ============================================
# 组合图形
# ============================================
p_final <- p2 +
  annotation_custom(
    grob = ggplotGrob(p3),                  
    xmin = 1350, xmax = 2800,                
    ymin = -1850, ymax = -1150               
  )
