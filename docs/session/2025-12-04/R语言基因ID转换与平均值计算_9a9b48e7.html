                       keys = ensembl_ids,
                       column = "SYMBOL",
                       keytype = "ENSEMBL",
                       multiVals = "first")

# 创建映射数据框
mapping <- data.frame(
    ensembl = ensembl_ids,
    symbol = gene_symbols,
    stringsAsFactors = FALSE
)

exp$symbol <- gene_symbols

# 移除没有映射到 symbol 的基因

# 对重复的 gene symbol 取平均值
exp_avg <- aggregate(. ~ symbol, data = exp, FUN = mean)

# 将 symbol 设为行名

exp_avg$symbol <- NULL
