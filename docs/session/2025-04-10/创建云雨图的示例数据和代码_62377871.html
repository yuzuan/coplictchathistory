# 云雨图(Raincloud Plot)创建教程

我将帮您创建一个云雨图并提供示例数据。云雨图结合了半小提琴图、箱线图和散点图，能够很好地展示数据分布。

circulans_data <- rnorm(20, mean = 5, sd = 1.5)  # 最低平均值

# 确保数据为正数并取整(因为这是计数数据)
subtills_data <- round(pmax(subtills_data, 1))
cereus_data <- round(pmax(cereus_data, 1))
megaterium_data <- round(pmax(megaterium_data, 1))
circulans_data <- round(pmax(circulans_data, 1))

# 创建宽格式数据框
df <- data.frame(

  cereus = cereus_data,
  megaterium = megaterium_data,

## 完整代码

以下是创建云雨图的完整代码：

`

# 加载必要的包
library(ggplot2)
library(tidyr)
library(dplyr)
library(gghalves)  # 用于半小提琴图
library(ggsignif)  # 用于显著性标记

# 读入数据
df <- read.csv('BGCs_numbers_per_genomes.csv', header = TRUE)

# 转换成长数据格式
df1 <- gather(df) %>%
  rename("Clade" = key, "value" = value)

# 设置颜色
my_color = c('#84a354', '#b15053', '#cf9a2c', '#7d7c7f')

# 排序
variable <- c('subtills', 'cereus', 'megaterium', 'circulans')
my_sort <- factor(variable, levels = variable)

# 显著性检验比较组合
my_comparisons <- list(c("subtills", "cereus"),
                       c("cereus", "megaterium"),
                       c("megaterium", "circulans"))

# 创建云雨图
ggplot(df1, aes(factor(Clade, levels = my_sort), value)) +
  # 半小提琴图部分
  geom_half_violin(aes(fill = factor(Clade, levels = my_sort)),
                   side = 'r',
                   position = position_nudge(x = .25, y = 0),
                   cex = 0.8) +
  # 箱线图部分
  geom_boxplot(fill = "white",
               width = 0.1, cex = 0.8,
               position = position_nudge(x = .25, y = 0)) +
  # 散点部分
  geom_jitter(aes(color = factor(Clade, levels = my_sort)), 
              size = 1.5,
              position = position_jitter(width = 0.1)) +
  # 显著性标记
  geom_signif(comparisons = my_comparisons, 
              map_signif_level = function(p) sprintf("p = %.2g", p), 
              step_increase = 0.05,
              y_position = c(23, 20, 17),
              textsize = 5, vjust = -0.2,
              tip_length = 0) +
  # 设置颜色
  scale_fill_manual(values = my_color) +
  scale_color_manual(values = my_color, guide = 'none') +
  # 设置主题和标签
  theme_test(base_size = 15) +
  labs(x = NULL, y = 'No.of BGCs / genome') +
  scale_y_continuous(limits = c(0, 25)) +
  theme(legend.title = element_blank(),
        axis.title = element_text(size = 21, face = "bold"),
        axis.text = element_text(color = 'black', size = 17, face = "bold"),
        axis.text.x = element_text(face = 'bold.italic'),
        legend.text = element_text(size = 14, face = "bold.italic"))

# 加载必要的包
library(circlize)
library(readxl)

# 读取数据
data <- readxl::read_xlsx("data.xlsx")

# 设置类别顺序
data$category <- factor(data$category, levels = c("Forks", "Crossings", "Length", "SurfArea", "AvgDiam", "RootVolume", "Tips"))

# 定义颜色
category_colors <- c(
  "Forks" = "#E41A1C",       # 红色
  "Crossings" = "#377EB8",   # 蓝色
  "Length" = "#4DAF4A",      # 绿色
  "SurfArea" = "#984EA3",    # 紫色
  "AvgDiam" = "#FF7F00",     # 橙色
  "RootVolume" = "#FFFF33",  # 黄色
  "Tips" = "#A65628"         # 棕色
)

# 定义分组颜色
colors <- c(
  "KB" = "#66C2A5",   # 青绿色
  "BC" = "#FC8D62",   # 橙色
  "BAc" = "#8DA0CB"   # 蓝紫色
)

# 清除之前的图形
circos.clear()

# 设置circlize参数
circos.par(start.degree = 90, 
           gap.degree = 7, 
           track.margin = c(0, 0.12), 
           cell.padding = c(0, 0, 0, 0)
)

# 初始化极坐标系
circos.initialize(factors = data$category, xlim = c(0, 1))

# 添加扇形分类标签和填充颜色
circos.track(
  factors = data$category, 
  ylim = c(0, 1), 
  track.height = 0.035, 
  bg.col = category_colors,
  panel.fun = function(x, y) {
    circos.text(
      CELL_META$xcenter,
      CELL_META$ylim[2] - 3.2,
      CELL_META$sector.index, 
      facing = "bending.inside",
