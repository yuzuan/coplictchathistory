## 1) 按基因合并（推荐：取交集，最稳）

# exp1, exp2, exp3: matrix 或 data.frame
exp1 <- as.matrix(exp1)
exp2 <- as.matrix(exp2)
exp3 <- as.matrix(exp3)

# 取三个数据集共同基因
genes <- Reduce(intersect, list(rownames(exp1), rownames(exp2), rownames(exp3)))

exp1i <- exp1[genes, , drop = FALSE]
exp2i <- exp2[genes, , drop = FALSE]
exp3i <- exp3[genes, , drop = FALSE]

# 按列拼接（样本拼接）
exp <- cbind(exp1i, exp2i, exp3i)

如果你想保留所有基因（并集）也可以做，但会出现缺失值，需要额外处理；交集最省心。

plot_pca <- function(mat, meta, title) {
  pca <- prcomp(t(mat), center = TRUE, scale. = TRUE)  # 对样本做PCA，所以要转置
  df <- data.frame(
    sample = rownames(pca$x),
    PC1 = pca$x[, 1],
    PC2 = pca$x[, 2]
  )
  df <- merge(df, meta, by = "sample", sort = FALSE)

  ggplot(df, aes(PC1, PC2, color = batch)) +
    geom_point(size = 2.5, alpha = 0.9) +
    labs(title = title, x = "PC1", y = "PC2") +
    theme_bw()

}

library(tidyr)

plot_sample_box <- function(mat, meta, title, n_genes = 2000, seed = 1) {
  set.seed(seed)
  genes <- rownames(mat)
  genes_use <- if (length(genes) > n_genes) sample(genes, n_genes) else genes

  df <- as.data.frame(t(mat[genes_use, , drop = FALSE]))
  df$sample <- rownames(df)
  df <- merge(df, meta, by = "sample", sort = FALSE)

  df_long <- pivot_longer(df, cols = all_of(genes_use), names_to = "gene", values_to = "value")

  ggplot(df_long, aes(x = sample, y = value, fill = batch)) +
    geom_boxplot(outlier.size = 0.2, width = 0.7) +
    labs(title = title, x = "Sample", y = "log2(TPM/CPM)") +
    theme_bw() +
    theme(axis.text.x = element_blank(), axis.ticks.x = element_blank())
}
