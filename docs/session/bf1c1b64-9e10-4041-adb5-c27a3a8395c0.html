# 封装函数，支持多种ID类型
ids_to_symbol <- function(ids) {
  ids <- unique(ids)
  mapped <- rep(NA_character_, length(ids))
  names(mapped) <- ids

  # 1. RefSeq (NM_/NR_/XM_/XR_/AK_)
  is_refseq <- grepl("^[NXA][MRK]_\\d+", ids)
  if (any(is_refseq)) {
    mapped[is_refseq] <- mapIds(org.Hs.eg.db,
                                keys = ids[is_refseq],
                                keytype = "REFSEQ",
                                column = "SYMBOL",
                                multiVals = "first")
  }

  # 2. ENSEMBL transcript (ENST)
  is_ensembl_trans <- grepl("^ENST\\d+", ids)
  if (any(is_ensembl_trans)) {
    mapped[is_ensembl_trans] <- mapIds(org.Hs.eg.db,
                                       keys = ids[is_ensembl_trans],
                                       keytype = "ENSEMBLTRANS",
                                       column = "SYMBOL",
                                       multiVals = "first")
  }

  # 3. ENSEMBL gene (ENSG)
  is_ensembl_gene <- grepl("^ENSG\\d+", ids)
  if (any(is_ensembl_gene)) {
    mapped[is_ensembl_gene] <- mapIds(org.Hs.eg.db,
                                      keys = ids[is_ensembl_gene],
                                      keytype = "ENSEMBL",
                                      column = "SYMBOL",
                                      multiVals = "first")
  }

  # 4. GenBank (BC开头等) 可选：用rentrez补充
  # 5. AceView、Havana transcript、UCSC Genes等可用biomaRt/TxDb补充
  # 这里只做ENST/ENSG/RefSeq，其他类型可按需要补充

  # 6. 其余当作已是symbol，原样保留
  is_symbol <- is.na(mapped)
  mapped[is_symbol] <- ids[is_symbol]

  # 去掉NA
  mapped <- mapped[!is.na(mapped)]

  return(mapped)
}
