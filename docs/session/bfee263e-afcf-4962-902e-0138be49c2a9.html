**如果使用 Python：**

import mygene

# 创建查询对象
mg = mygene.MyGeneInfo()

# 转换 GB_ACC 到 gene symbol
results = mg.querymany(
    anno['ID'].tolist(),
    scopes='refseq',
    fields='symbol',
    species='human'
)

# 提取 gene symbol
gene_map = {r['query']: r.get('symbol', 'NA') for r in results}
anno['gene_symbol'] = anno['ID'].map(gene_map)
