# 目录路径
dir_path <- "/Users/yuzuan/Downloads/Downloads/2026_01_30 GSE156928_RAW"

# 收集所有文件
files <- list.files(
  dir_path,
  pattern = "\\.tar\\.gz$|_quant\\.sf\\.txt\\.gz$",
  full.names = TRUE
)

read_one <- function(f) {
  sample <- sub("^(GSM\\d+).*", "\\1", basename(f))  # 只取 GSM 开头的样本名
  
  if (grepl("\\.tar\\.gz$", f)) {
    paths <- utils::untar(f, list = TRUE)
    qpath <- paths[grepl("quant\\.genes\\.sf$", paths)]
    if (length(qpath) == 0) stop("No quant.genes.sf in ", f)
    tmpdir <- tempfile()
    dir.create(tmpdir)
    utils::untar(f, files = qpath[1], exdir = tmpdir)
    df <- read.table(
      file.path(tmpdir, qpath[1]),
      header = TRUE, sep = "\t", quote = "",
      comment.char = "", stringsAsFactors = FALSE
    )
  } else {
    df <- read.table(
      gzfile(f),
      header = TRUE, sep = "\t", quote = "",
      comment.char = "", stringsAsFactors = FALSE
    )
  }
  
  data.frame(
    gene = df$Name,
    setNames(df$TPM, sample),
    check.names = FALSE,
    stringsAsFactors = FALSE
  )
}

# 读取并合并
dfs <- lapply(files, read_one)
expr <- Reduce(function(x, y) merge(x, y, by = "gene", all = TRUE), dfs)

# 行名为 gene，列名为样本
rownames(expr) <- expr$gene
expr_mat <- expr
expr_mat$gene <- NULL

# 保存（gene 作为第一列，保证 R 可准确读取）
out_file <- file.path(dir_path, "GSE156928_gene_TPM_matrix.tsv")
write.table(
  data.frame(gene = rownames(expr_mat), expr_mat, check.names = FALSE),
  file = out_file, sep = "\t", quote = FALSE, row.names = FALSE
)
