我来帮你创建一个小提琴图，展示上调和下调基因各top30的表达情况。

# 获取上调和下调基因各top30
DEG_up <- DEG[DEG$change == "Up", ]
DEG_up <- DEG_up[order(DEG_up$logFC, decreasing = TRUE), ]
top30_up <- head(DEG_up, 30)

DEG_down <- DEG[DEG$change == "Down", ]
DEG_down <- DEG_down[order(DEG_down$logFC, decreasing = FALSE), ]
top30_down <- head(DEG_down, 30)

# 合并top30基因
top_genes <- rbind(top30_up, top30_down)

# 提取这些基因的表达数据
exp_top <- exp[top_genes$gene, ]

# 转换为长格式数据框
library(reshape2)
exp_long <- melt(as.matrix(exp_top))
colnames(exp_long) <- c("Gene", "Sample", "Expression")

# 添加分组信息
exp_long$Group <- group[match(exp_long$Sample, colnames(exp_top))]

# 添加基因变化类型
exp_long$Change <- top_genes$change[match(exp_long$Gene, top_genes$gene)]

# 绘制小提琴图
library(ggplot2)

# 方法1: 按基因分组的小提琴图
p1 <- ggplot(exp_long, aes(x = Gene, y = Expression, fill = Group)) +
  geom_violin(trim = FALSE, scale = "width") +
  geom_boxplot(width = 0.1, position = position_dodge(0.9), outlier.size = 0.5) +
  facet_wrap(~ Change, scales = "free_x", ncol = 1) +
  theme_bw() +
  theme(axis.text.x = element_text(angle = 90, hjust = 1, vjust = 0.5, size = 6),
        legend.position = "top") +
  labs(title = "Top 30 Up/Down-regulated Genes Expression",
       x = "Gene", y = "Expression Level") +
  scale_fill_manual(values = c("Control" = "#4DBBD5", "VD" = "#E64B35"))

ggsave(file.path(out_dir, "violin_top30_by_gene.pdf"), p1, width = 14, height = 10)

# 方法2: 按变化类型汇总的小提琴图（更简洁）
p2 <- ggplot(exp_long, aes(x = Change, y = Expression, fill = Group)) +
  geom_violin(trim = FALSE, alpha = 0.8) +
  geom_boxplot(width = 0.15, position = position_dodge(0.9), outlier.size = 0.5) +
  theme_bw() +
  theme(legend.position = "top",
        axis.text = element_text(size = 12),
        axis.title = element_text(size = 14)) +
  labs(title = "Expression Distribution of Top 30 DEGs",
       x = "Regulation Type", y = "Expression Level") +
  scale_fill_manual(values = c("Control" = "#4DBBD5", "VD" = "#E64B35"))

ggsave(file.path(out_dir, "violin_top30_summary.pdf"), p2, width = 6, height = 6)

print(p1)
print(p2)

这段代码会生成两种小提琴图：

1. **violin_top30_by_gene.pdf** - 每个基因单独展示，按Up/Down分面显示
2. **violin_top30_summary.pdf** - 汇总展示，更简洁地比较上调和下调基因在两组间的表达差异

主要特点：
- 使用 logFC 排序选取top30（上调取logFC最大的，下调取logFC最小的）
- 小提琴图内嵌箱线图，更好展示数据分布
