library(ggplot2)
library(ggrepel)

# 准备数据
diff <- DEG
diff$external_gene_name <- rownames(diff)
diff$FDR <- diff$P.Value

# 提取上调和下调基因
diff_up <- diff[diff$change == "UP", ]
diff_down <- diff[diff$change == "DOWN", ]

# 准备背景点（非显著基因）
data_bg <- diff[diff$change == "NOT", ]

# 准备基因集数据（显著差异基因）
data_geneset <- rbind(diff_up, diff_down)
data_geneset$gene_col <- ifelse(data_geneset$change == "UP", "Up", "Down")

# 选择关键基因进行标注（Top 10 上调 + Top 10 下调）
data_key <- rbind(
head(diff_up[order(diff_up$P.Value), ], 10),
head(diff_down[order(diff_down$P.Value), ], 10)
)
data_key$gene_col <- "key"

# 绘制火山图
p <- ggplot(diff, aes(x = logFC, y = -log10(FDR))) +
# 上调基因显著性阴影区域
annotate(
geom = "rect", fill = "#ea838e", alpha = 0.3,
xmin = min(diff_up$logFC), xmax = max(diff_up$logFC),
ymin = -log10(pvalue_t), ymax = max(-log10(diff$FDR))
) +
# 下调基因显著性阴影区域
annotate(
geom = "rect", fill = "#827fd6", alpha = 0.3,
xmin = min(diff_down$logFC), xmax = max(diff_down$logFC),
ymin = -log10(pvalue_t), ymax = max(-log10(diff$FDR))
) +
# 背景灰色点（非显著基因）
geom_point(data = data_bg, shape = 21, color = "black",
alpha = 0.2, size = 1.3, stroke = 0.6) +
# 显著差异基因点
geom_point(data = rbind(data_geneset, data_key),
aes(fill = gene_col), shape = 21, color = "black",
size = 2, stroke = 0.8) +
# 设置颜色
scale_fill_manual(values = c("Up" = "#faa400",
"Down" = "#bebebe",
"key" = "#0afdfe")) +
# 标注关键基因名称
geom_label_repel(
data = data_key, aes(label = external_gene_name),
size = 6, fill = "white", box.padding = unit(0.35, "lines"),
alpha = 0.8, point.padding = 0.5,
segment.colour = "#4c4b5e", segment.size = 0.5,
min.segment.length = 0
) +
# 添加参考线
geom_vline(xintercept = 0, color = "#b2b2b2", linewidth = 0.6) +
geom_hline(yintercept = 0, color = "#b2b2b2", linewidth = 0.6) +
geom_hline(yintercept = -log10(pvalue_t), color = "#ff7d82",
linewidth = 0.6, linetype = "dotted") +
# 设置坐标轴范围
scale_x_continuous(limits = c(-3, 3), breaks = seq(-3, 3, 1)) +
# 主题设置
theme_classic() +
theme(
legend.position = "none",
axis.title.x = element_text(size = 16),
axis.title.y = element_text(size = 16),
axis.text.x = element_text(size = 12),
axis.text.y = element_text(size = 12)
) +
labs(
x = expression(log[2]("Fold Change")),
y = expression(-log[10]("FDR"))
)

# 显示图形
print(p)

# 保存图形
custom_volcano_pdf <- paste0(out_dir, "/Custom_Volcano_Plot.pdf")
ggsave(custom_volcano_pdf, p, width = 10, height = 8)
cat("✓ 自定义火山图已保存:", basename(custom_volcano_pdf), "\n")
