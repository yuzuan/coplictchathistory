        })

    ref_idx <- grepl('^[NXA][MRK]_', ids)

    }

    ensmust_idx <- grepl('^ENSMUST', ids)

        if (verbose) cat('🔄 ENSMUST -> SYMBOL\n')
        tmp <- convert_ensmust(ids[ensmust_idx], mart = mart)
        mapped[ensmust_idx] <- tmp
        source[ensmust_idx][!is.na(tmp) & tmp != ""] <- "ENSMUST"
    }
    
    # ENSMUSG转换 (小鼠Ensembl基因)
    ensmusg_idx <- grepl('^ENSMUSG', ids)
    if (any(ensmusg_idx)) {
        if (verbose) cat('🔄 ENSMUSG -> SYMBOL\n')
        tmp <- convert_ensmusg(ids[ensmusg_idx], mart = mart)
        mapped[ensmusg_idx] <- tmp
        source[ensmusg_idx][!is.na(tmp) & tmp != ""] <- "ENSMUSG"
    }
    
    # OTTMUST转换 (Havana小鼠转录本)
    ottmust_idx <- grepl('^OTTMUST', ids)
    if (any(ottmust_idx)) {
        if (verbose) cat('🔄 OTTMUST -> SYMBOL\n')
        tmp <- convert_ottmust(ids[ottmust_idx], gtf_mapping = gtf_mapping, mart = mart)
        mapped[ottmust_idx] <- tmp
        source[ottmust_idx][!is.na(tmp)] <- "OTTMUST"

        tmp <- convert_mgi(ids[mgi_idx])
        mapped[mgi_idx] <- tmp
        source[mgi_idx][!is.na(tmp)] <- "MGI"

    entrez_idx <- grepl('^\\d+$', ids) & is.na(mapped)

        tmp <- convert_entrez_mouse(ids[entrez_idx])
        mapped[entrez_idx] <- tmp
        source[entrez_idx][!is.na(tmp)] <- "Entrez"

        ID = orig_ids,
        SYMBOL = mapped,
        Source = source
    )

比较方向 experimental vs control 的含义：

| logFC | 含义 |
|-------|------|
| logFC > 0 | 基因在 **experimental** 中表达 **高于** control → **上调 (Up)** |
| logFC < 0 | 基因在 **experimental** 中表达 **低于** control → **下调 (Down)** |

所以：
- **Up** = experimental 组相对于 control 组上调
