# 提取第一个基因名（格式：xxx // GeneSymbol // ...）
extract_gene <- function(x) {
  if (x == "---" || is.na(x)) return(NA)
  # 提取第一个 // 后面的基因名
  parts <- strsplit(x, " // ")[[1]]
  if (length(parts) >= 2) return(parts[2])
  return(NA)
}
